  <script>
      document.addEventListener("DOMContentLoaded", () => {
  // Reveal on scroll
  const sections = document.querySelectorAll("section, header.hero");
  const reveal = () => {
    sections.forEach(sec => {
      const rect = sec.getBoundingClientRect();
      if (rect.top < window.innerHeight * 0.85) {
        sec.style.opacity = 1;
        sec.style.transform = "translateY(0)";
      }
    });
  };
  sections.forEach(sec => {
    sec.style.opacity = 0;
    sec.style.transform = "translateY(50px)";
    sec.style.transition = "all 0.6s ease-out";
  });
  window.addEventListener("scroll", reveal, { passive: true });
  reveal();
  });


    // Mobile menu toggle
    const menuBtn = document.getElementById('menuBtn');
    const mobileMenu = document.getElementById('mobileMenu');
    menuBtn.addEventListener('click', () => {
      const open = mobileMenu.style.display === 'block';
      mobileMenu.style.display = open ? 'none' : 'block';
      menuBtn.setAttribute('aria-label', open ? 'Open Menu' : 'Close Menu');
    });

    // Intersection Observer reveal
    const io = new IntersectionObserver((entries)=>{
      entries.forEach(e=>{if(e.isIntersecting){e.target.classList.add('in');io.unobserve(e.target)}})
    },{threshold:.12});
    document.querySelectorAll('.reveal').forEach(el=>io.observe(el));

    // Quote calculator
    const quoteForm = document.getElementById('quoteForm');
    const totalEl = document.getElementById('quoteTotal');
    const estimateField = document.getElementById('estimateField');
    const sendQuoteBtn = document.getElementById('sendQuote');

    function formatUSD(n){return new Intl.NumberFormat('en-US',{style:'currency',currency:'USD'}).format(n)}
    function calcQuote(){
      const fd = new FormData(quoteForm); let total = 0; let monthly = 0;
      for (const [k,v] of fd.entries()){
        const n = Number(v)||0; total += n;
        const field = quoteForm.querySelector(`[name="${k}"]`);
        if(field && field.hasAttribute('data-recurring')) monthly += n;
      }
      const text = monthly>0? `${formatUSD(total)} + ${formatUSD(monthly)}/mo` : formatUSD(total);
      totalEl.textContent = text; estimateField.value = text;
    }
    quoteForm.addEventListener('change', calcQuote);
    calcQuote();

    // Send quote to message textarea
    sendQuoteBtn.addEventListener('click', ()=>{
      const msg = document.getElementById('message');
      const current = msg.value.trim();
      const add = `\n\nEstimated package total: ${estimateField.value}`;
      msg.value = current ? current + add : `Hi Dylan, I'd like to start a project. ${add}`;
      document.getElementById('contact').scrollIntoView({behavior:'smooth'});
      msg.focus();
    });

    // Year in footer
    document.getElementById('yr').textContent = new Date().getFullYear();

    // Close mobile menu on link click
    document.querySelectorAll('.mobile a').forEach(a=>a.addEventListener('click',()=>{mobileMenu.style.display='none'}));

      var form = document.getElementById("my-form");
  // funtion to send msg from contact form
  async function handleSubmit(event) {
    event.preventDefault();
    var status = document.getElementById("my-form-status");
    var data = new FormData(event.target);
    fetch(event.target.action, {
      method: form.method,
      body: data,
      headers: {
          'Accept': 'application/json'
      }
    }).then(response => {
      if (response.ok) {
        status.innerHTML = "Thanks for your submission!";
        form.reset()
      } else {
        response.json().then(data => {
          if (Object.hasOwn(data, 'errors')) {
            status.innerHTML = data["errors"].map(error => error["message"]).join(", ")
          } else {
            status.innerHTML = "Oops! There was a problem submitting your form"
          }
        })
      }
    }).catch(error => {
      status.innerHTML = "Oops! There was a problem submitting your form"
    });
  }
  form.addEventListener("submit", handleSubmit)
  </script>
